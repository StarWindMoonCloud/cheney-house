f(s1,s2,s3) = (f(s1[0,1,...,m-1],s2, s3[0,1,...,m+n-1]) || f(s1,s2,s3) = f(s1,s2[0,1,...,n-1], s3[0,1,...,m+n-1]))
d. s3[m+n] != s1[m] && s3[m+n] != s2[n]
