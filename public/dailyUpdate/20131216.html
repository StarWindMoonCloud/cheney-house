    void dfs(int start, string s, vector &path, vector > &res) {
        if (start >= s.length()) {
            // find an answer
            res.push_back(path);
            return;
        }
        for (int i = start; i < s.length(); i++) {
            string s1 = s.substr(start, i-start+1); // start to i                                                                                             
            if (isP(s1)) {
                path.push_back(s1);
                dfs(i+1, s, path, res);
                path.pop_back();
            }
        }
    }
